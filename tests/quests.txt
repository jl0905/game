# Quests (direction F4): a bandit hunt taken in town pays out on the kill;
# a grain delivery pays at the destination gate.
seed 5
enter 0            # Sargoth, day 0 -> the giver offers the Bandit Hunt
quest
state              # expect quest: hunt progress=0
leave
hunt 120           # ride down the nearest outlaw band
block on
tick 150
state              # expect QUEST COMPLETE (+100 gold on top of loot)
enter 3            # Jelkala, day 0 -> the giver offers the Grain Delivery
quest
state              # expect quest: grain target=8 (Emberfall)
leave
enter 8            # Emberfall: buy the cargo at the source
market
buy grain 5
market
leave
enter 8            # arriving at the destination gate pays out
state              # expect QUEST COMPLETE and grain consumed
quit
